#include <stdio.h>
#include <string.h>

char tracks[][80] = {
        "I left my heart in Haevard Med School",
        "Newark, Newark - a wonderful town",
        "Dancing with a Dork",
        "From here to maternity",
        "The girl from Iwo Jima",
};

void name(char name[])
{
    int i;
    for (i = 0; i < 5; i++){
        if (strstr(tracks[i], name))
            printf("Track %i: %s\n", i + 1, tracks[i]);
    }
}

void findTrack(char search[])
{
    int i;
    for (i = 0; i < 5; i++){
        if (strstr(tracks[i], search))
            printf("Track %i: '%s'\n", i + 1, tracks[i]);
    }
    printf("%s\n", search);
}

int main()
{
    char searchFor[80];
    name("town");
    printf("Search for: ");
    fgets(searchFor, 80, stdin);
    findTrack(searchFor);
    return 0;
}
